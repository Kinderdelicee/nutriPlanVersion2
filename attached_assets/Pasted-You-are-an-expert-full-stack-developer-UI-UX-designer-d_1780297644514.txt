Full Prisma Schema (Create Exactly This)
prismagenerator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = "file:./dev.db"
}

model User {
  id            String    @id @default(cuid())
  email         String    @unique
  password      String
  name          String
  age           Int?
  gender        String?
  weight        Float?    // kg
  height        Float?    // cm
  activityLevel String?   // Sedentary, Light, Moderate, Active, Very Active
  dailyCalories Int?      @default(2000)
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  foodLogs      FoodLog[]
  meals         Meal[]
  savedMeals    SavedMeal[]
}

model Food {
  id          String   @id @default(cuid())
  name        String
  calories    Int
  protein     Float    @default(0)
  carbs       Float    @default(0)
  fat         Float    @default(0)
  servingSize String   @default("100g")
  category    String?
  isCustom    Boolean  @default(false)
  userId      String?
  createdAt   DateTime @default(now())

  foodLogs    FoodLog[]
  mealItems   MealItem[]
}

model Meal {
  id            String     @id @default(cuid())
  userId        String
  name          String
  description   String?
  totalCalories Int
  totalProtein  Float
  totalCarbs    Float
  totalFat      Float
  createdAt     DateTime   @default(now())

  user          User       @relation(fields: [userId], references: [id])
  items         MealItem[]
}

model MealItem {
  id       String @id @default(cuid())
  mealId   String
  foodId   String
  quantity Float
  calories Int
  protein  Float
  carbs    Float
  fat      Float

  meal     Meal   @relation(fields: [mealId], references: [id])
  food     Food   @relation(fields: [foodId], references: [id])
}

model FoodLog {
  id             String   @id @default(cuid())
  userId         String
  date           DateTime
  mealType       String   // Breakfast, Lunch, Dinner, Snack
  foodId         String?
  customFoodName String?
  quantity       Float
  calories       Int
  protein        Float
  carbs          Float
  fat            Float
  notes          String?

  user           User     @relation(fields: [userId], references: [id])
  food           Food?    @relation(fields: [foodId], references: [id])
}

model SavedMeal {
  id     String @id @default(cuid())
  userId String
  mealId String

  user   User   @relation(fields: [userId], references: [id])
  meal   Meal   @relation(fields: [mealId], references: [id])
}
Run npx prisma generate and npx prisma db push after creating the schema.
